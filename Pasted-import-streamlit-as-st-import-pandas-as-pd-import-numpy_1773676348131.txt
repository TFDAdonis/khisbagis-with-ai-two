import re as _re

def _parse_float(text, pattern, default=None):
    m = _re.search(pattern, text)
    if m:
        try:
            return float(m.group(1))
        except Exception:
            pass
    return default

def _parse_floats_list(text, pattern):
    m = _re.search(pattern, text)
    if m:
        try:
            return [float(x) for x in _re.findall(r'[-\d.]+', m.group(1)) if x not in ('', '-')]
        except Exception:
            pass
    return []

def get_smart_interpretation(chart_type, data_summary, location=""):
    """Generate an instant, data-driven interpretation for any chart type."""
    ct = chart_type.lower()
    loc_str = f" for {location}" if location else ""

    # ── Climate Classification ──────────────────────────────────────────────
    if "climate classification" in ct or "climate zone" in ct:
        temp = _parse_float(data_summary, r'Mean temperature:\s*([\d.]+)')
        precip = _parse_float(data_summary, r'Annual precipitation:\s*([\d.]+)')
        zone = ""
        zm = _re.search(r'Climate zone:\s*([^,]+)', data_summary)
        if zm:
            zone = zm.group(1).strip()
        aridity = _parse_float(data_summary, r'Aridity index:\s*([\d.]+)')

        parts = []
        if zone:
            parts.append(f"The climate{loc_str} is classified as **{zone}**.")
        if temp is not None:
            if temp > 30:
                parts.append(f"With a mean annual temperature of {temp:.1f}°C, heat stress is a significant factor for crop selection — drought-tolerant and heat-adapted varieties are recommended.")
            elif temp > 20:
                parts.append(f"A mean annual temperature of {temp:.1f}°C supports year-round cultivation of warm-season crops such as maize, sorghum, and legumes.")
            elif temp > 10:
                parts.append(f"A mean annual temperature of {temp:.1f}°C is ideal for temperate crops including wheat, barley, and a wide range of vegetables.")
            else:
                parts.append(f"A mean annual temperature of {temp:.1f}°C limits the growing season; cold-hardy crops and frost management are critical.")
        if precip is not None:
            if precip < 250:
                parts.append(f"Annual precipitation of {precip:.0f} mm indicates a hyper-arid regime — irrigation is essential for any agricultural production.")
            elif precip < 500:
                parts.append(f"Annual precipitation of {precip:.0f} mm is semi-arid; supplemental irrigation during dry spells will significantly improve yields.")
            elif precip < 800:
                parts.append(f"Annual precipitation of {precip:.0f} mm supports rainfed agriculture for most of the year, though seasonal deficits may require supplemental irrigation.")
            else:
                parts.append(f"High annual precipitation of {precip:.0f} mm means waterlogging and fungal diseases may need attention; drainage management is advisable.")
        if aridity is not None:
            if aridity < 0.2:
                parts.append(f"The aridity index of {aridity:.2f} confirms extreme water scarcity conditions.")
            elif aridity < 0.5:
                parts.append(f"An aridity index of {aridity:.2f} places this area in the semi-arid category with seasonal moisture stress.")
        return " ".join(parts) if parts else "Climate classification indicates typical regional conditions suitable for adapted local crop varieties."

    # ── Temperature ─────────────────────────────────────────────────────────
    if "temperature" in ct:
        temps = _parse_floats_list(data_summary, r'Monthly temps?:\s*(\[[\d.,\s-]+\])')
        max_t = _parse_float(data_summary, r'Max(?:imum)?(?:\s*Temp(?:erature)?)?\s*[=:]?\s*([\d.]+)°?C')
        min_t = _parse_float(data_summary, r'Min(?:imum)?(?:\s*Temp(?:erature)?)?\s*[=:]?\s*([\d.]+)°?C')
        max_m = ""
        mm = _re.search(r'Max.*?in\s+(\w+)', data_summary)
        if mm:
            max_m = mm.group(1)
        min_m = ""
        mnm = _re.search(r'Min.*?in\s+(\w+)', data_summary)
        if mnm:
            min_m = mnm.group(1)

        if not temps and max_t and min_t:
            temps = [min_t, max_t]

        parts = []
        if max_t and min_t:
            rng = max_t - min_t
            parts.append(f"Temperatures{loc_str} span {min_t:.1f}°C to {max_t:.1f}°C — a seasonal range of {rng:.1f}°C.")
            if max_m:
                parts.append(f"Peak heat in {max_m} represents the critical summer stress period.")
            if min_m:
                parts.append(f"The coolest month is {min_m}.")
            if rng > 25:
                parts.append("Such a large annual range is typical of continental climates; frost protection in winter and heat management in summer are both priorities.")
            elif rng > 15:
                parts.append("A moderate seasonal swing allows for diverse crop rotations — warm-season crops in summer and cool-season crops in autumn/spring.")
            else:
                parts.append("The mild seasonal variation supports year-round cultivation with minimal frost risk.")
        if temps:
            avg = sum(temps) / len(temps)
            hot_months = sum(1 for t in temps if t > 30)
            cold_months = sum(1 for t in temps if t < 5)
            if hot_months > 0:
                parts.append(f"{hot_months} month(s) exceed 30°C — irrigation and shade management are recommended during these periods.")
            if cold_months > 0:
                parts.append(f"{cold_months} month(s) fall below 5°C, indicating frost risk for sensitive crops.")
        return " ".join(parts) if parts else f"Temperature data{loc_str} shows typical seasonal patterns for this climate zone."

    # ── Precipitation / Water ───────────────────────────────────────────────
    if "precipitation" in ct or "water" in ct or "evapotranspir" in ct:
        annual = _parse_float(data_summary, r'Annual total:\s*([\d.]+)')
        peak_m = ""
        pm = _re.search(r'Peak month:\s*(\w+)', data_summary)
        if pm:
            peak_m = pm.group(1)
        precips = _parse_floats_list(data_summary, r'Monthly precipitation.*?:\s*(\[[\d.,\s]+\])')

        parts = []
        if annual is not None:
            if annual < 200:
                parts.append(f"Total annual precipitation{loc_str} is extremely low at {annual:.0f} mm, placing this location in a hyper-arid category where agriculture is impossible without intensive irrigation.")
            elif annual < 400:
                parts.append(f"Annual rainfall of {annual:.0f} mm{loc_str} is scarce — efficient drip irrigation, water harvesting, and drought-tolerant crop varieties are essential for viable production.")
            elif annual < 700:
                parts.append(f"Annual precipitation of {annual:.0f} mm{loc_str} can support rainfed agriculture for most crops, though summer deficits likely require supplemental irrigation.")
            else:
                parts.append(f"Generous annual rainfall of {annual:.0f} mm{loc_str} supports productive rainfed farming; however, waterlogging and erosion management remain important.")
        if peak_m:
            parts.append(f"Rainfall peaks in {peak_m} — this is the optimal planting window for rain-fed crops.")
        if precips:
            dry_months = sum(1 for p in precips if p < 20)
            if dry_months > 3:
                parts.append(f"{dry_months} months receive less than 20 mm, indicating a pronounced dry season where crops will need water supplementation.")
        return " ".join(parts) if parts else f"Precipitation data{loc_str} shows a typical seasonal distribution for this region."

    # ── Soil Moisture by Layer ──────────────────────────────────────────────
    if "soil moisture" in ct and ("layer" in ct or "distribution" in ct or "comparison" in ct):
        surf = _parse_float(data_summary, r'(?:Mean )?[Ss]urface.*?:\s*([\d.]+)\s*m')
        root = _parse_float(data_summary, r'[Rr]oot.*?zone.*?:\s*([\d.]+)\s*m')
        deep = _parse_float(data_summary, r'[Dd]eep.*?:\s*([\d.]+)\s*m')
        wet_m = ""
        wm = _re.search(r'Wettest month:\s*(\w+)', data_summary)
        if wm:
            wet_m = wm.group(1)

        parts = []
        if surf is not None and root is not None and deep is not None:
            if surf > 0.3:
                parts.append(f"Surface soil moisture{loc_str} is high at {surf:.3f} m³/m³, indicating recent rainfall or waterlogging risk in poorly drained soils.")
            elif surf > 0.15:
                parts.append(f"Surface soil moisture of {surf:.3f} m³/m³{loc_str} is moderate — adequate for germination and shallow-rooted crops.")
            else:
                parts.append(f"Low surface soil moisture ({surf:.3f} m³/m³){loc_str} indicates dry topsoil conditions; timely irrigation is needed for seedling establishment.")

            if root is not None:
                if root > 0.25:
                    parts.append(f"Root-zone moisture ({root:.3f} m³/m³) is well-supplied, supporting active crop growth without irrigation stress.")
                elif root > 0.1:
                    parts.append(f"Root-zone moisture ({root:.3f} m³/m³) is marginal — crops with moderate water demand should be prioritized.")
                else:
                    parts.append(f"Root-zone moisture ({root:.3f} m³/m³) is critically low, which will severely limit plant growth and productivity.")

            if root and surf:
                gradient = surf - root
                if gradient > 0.05:
                    parts.append("The pronounced decrease with depth suggests high evaporation at the surface and limited deep percolation.")
                elif gradient < -0.05:
                    parts.append("Higher moisture in the root zone than at the surface may indicate capillary rise from a shallow water table.")
        if wet_m:
            parts.append(f"Peak moisture occurs in {wet_m}, aligning with the main growing season.")
        return " ".join(parts) if parts else f"Soil moisture profile{loc_str} shows typical layered distribution for the regional soil type."

    # ── Soil Texture ────────────────────────────────────────────────────────
    if "soil texture" in ct:
        clay = _parse_float(data_summary, r'Clay:\s*([\d.]+)%')
        silt = _parse_float(data_summary, r'Silt:\s*([\d.]+)%')
        sand = _parse_float(data_summary, r'Sand:\s*([\d.]+)%')
        tex = ""
        tm = _re.search(r'Texture class:\s*([^,\n]+)', data_summary)
        if tm:
            tex = tm.group(1).strip()

        parts = []
        if tex:
            parts.append(f"The soil{loc_str} is classified as **{tex}**.")
        if clay is not None and sand is not None and silt is not None:
            if clay > 40:
                parts.append(f"High clay content ({clay:.0f}%) provides excellent nutrient and water retention but requires careful tillage to prevent compaction and waterlogging.")
            elif clay > 25:
                parts.append(f"Moderate clay content ({clay:.0f}%) offers good structure — this soil holds nutrients well and is suitable for a wide range of crops.")
            elif sand > 60:
                parts.append(f"Sandy texture ({sand:.0f}% sand) means rapid drainage and low water retention; frequent irrigation and organic matter additions are recommended.")
            else:
                parts.append(f"A balanced texture ({clay:.0f}% clay, {silt:.0f}% silt, {sand:.0f}% sand) supports good soil structure, tilth, and workability.")
            if silt > 40:
                parts.append(f"High silt content ({silt:.0f}%) improves moisture retention and is beneficial for arable crops, though erosion risk is elevated on slopes.")
        return " ".join(parts) if parts else f"Soil texture{loc_str} indicates typical properties for the region."

    # ── Soil Organic Matter ─────────────────────────────────────────────────
    if "organic matter" in ct or "som" in ct:
        som = _parse_float(data_summary, r'Soil Organic Matter:\s*([\d.]+)%')
        soc = _parse_float(data_summary, r'SOC Stock:\s*([\d.]+)\s*t')
        bd = _parse_float(data_summary, r'Bulk density:\s*([\d.]+)')
        depth = _parse_float(data_summary, r'Depth:\s*([\d.]+)cm')

        parts = []
        if som is not None:
            if som < 1.0:
                parts.append(f"Soil organic matter{loc_str} is critically low at {som:.2f}% — this seriously limits soil fertility, water-holding capacity, and biological activity.")
                parts.append("Compost additions, cover cropping, and minimal tillage are urgently recommended to rebuild SOM.")
            elif som < 2.0:
                parts.append(f"SOM of {som:.2f}%{loc_str} is below optimal levels. Incremental organic inputs such as manure, green manures, or compost will improve soil health over time.")
            elif som < 4.0:
                parts.append(f"SOM of {som:.2f}%{loc_str} is in the moderate range — a good foundation for productive agriculture. Maintaining or slightly increasing this level through conservation practices is advisable.")
            else:
                parts.append(f"Excellent SOM of {som:.2f}%{loc_str} reflects a highly fertile, biologically active soil. This supports superior water retention, nutrient cycling, and crop yields.")
        if soc is not None:
            parts.append(f"The SOC stock of {soc:.1f} t/ha represents a significant carbon reservoir — protecting this through reduced tillage and residue retention has both agronomic and climate benefits.")
        if bd is not None:
            if bd > 1.5:
                parts.append(f"A bulk density of {bd} g/cm³ indicates compaction risk — subsoiling or deep tillage may be beneficial.")
            elif bd < 1.1:
                parts.append(f"A low bulk density of {bd} g/cm³ indicates a loose, well-aerated structure ideal for root development.")
        return " ".join(parts) if parts else f"Soil organic matter data{loc_str} indicates typical conditions for this soil type."

    # ── Vegetation Indices ──────────────────────────────────────────────────
    if any(v in ct for v in ['ndvi', 'evi', 'savi', 'ndwi', 'gndvi', 'msavi', 'osavi', 'arvi', 'vari', 'ndmi', 'nbr', 'mndwi', 'awei', 'ndsi', 'bri', 'mtvi', 'rdvi', 'nli', 'gdvi', 'vegetation index']):
        idx_m = _re.search(r'^(\w+)\s+vegetation', ct) or _re.search(r'^(\w+)\b', ct)
        idx_name = idx_m.group(1).upper() if idx_m else "Index"
        mean_v = _parse_float(data_summary, r'mean=([\d.]+)')
        max_v = _parse_float(data_summary, r'max=([\d.]+)')
        min_v = _parse_float(data_summary, r'min=([\d.]+)')
        trend_dir = "stable"
        td = _re.search(r'trend=(\w+)', data_summary)
        if td:
            trend_dir = td.group(1)
        trend_v = _parse_float(data_summary, r'trend=\w+\s*\(([-\d.]+)/month\)')

        parts = []
        if idx_name in ('NDVI', 'EVI', 'SAVI', 'MSAVI', 'OSAVI', 'GNDVI'):
            label = "vegetation greenness/biomass"
            if mean_v is not None:
                if mean_v > 0.6:
                    parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} — indicating dense, healthy vegetation cover with high photosynthetic activity.")
                elif mean_v > 0.4:
                    parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} — representing moderate vegetation cover typical of mixed cropland or savanna.")
                elif mean_v > 0.2:
                    parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} — sparse vegetation, possibly degraded rangeland, semi-arid scrub, or fallow fields.")
                else:
                    parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} — very low greenness consistent with bare soil, desert, or heavily stressed vegetation.")
        elif idx_name in ('NDWI', 'MNDWI', 'AWEI'):
            label = "water content / surface water"
            if mean_v is not None:
                if mean_v > 0.3:
                    parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} — high water content indicating well-irrigated fields or proximity to water bodies.")
                elif mean_v > 0:
                    parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} — moderate moisture, suitable conditions for most crops without significant water stress.")
                else:
                    parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} — indicates moisture deficit; irrigation scheduling should be based on regular monitoring.")
        elif idx_name == 'NBR':
            if mean_v is not None:
                if mean_v > 0.3:
                    parts.append(f"NBR{loc_str} of {mean_v:.3f} suggests healthy, unburned vegetation with good near-infrared reflectance.")
                else:
                    parts.append(f"NBR{loc_str} of {mean_v:.3f} may indicate post-fire recovery areas or stressed vegetation requiring monitoring.")
        else:
            if mean_v is not None:
                parts.append(f"{idx_name}{loc_str} averages {mean_v:.3f} (range: {min_v:.3f}–{max_v:.3f}).")

        if trend_dir == "increasing":
            rate = abs(trend_v) if trend_v else 0
            parts.append(f"A positive trend of {rate:.4f}/month indicates improving vegetation conditions — possibly responding to seasonal rainfall or improved land management.")
        elif trend_dir == "decreasing":
            rate = abs(trend_v) if trend_v else 0
            parts.append(f"A declining trend of {rate:.4f}/month is concerning — this may signal vegetation degradation, overgrazing, drought stress, or land use change.")
        else:
            parts.append("The index shows a stable trajectory over the monitoring period — no significant land cover change detected.")

        if max_v and min_v:
            rng = max_v - min_v
            if rng > 0.3:
                parts.append(f"High variability (range {rng:.3f}) reflects strong seasonal dynamics or field-to-field heterogeneity in the area.")
        return " ".join(parts) if parts else f"{idx_name} data{loc_str} indicates typical vegetation dynamics for this region."

    # ── Generic fallback ────────────────────────────────────────────────────
    return f"Analysis of {chart_type}{loc_str}: {data_summary[:200]}. Values are within expected ranges for this region. Monitor seasonal trends for anomaly detection."

def get_region_type(location_name):
    if not location_name:
        return "general"
    loc = location_name.lower()
    if any(x in loc for x in ['sidi', 'algeria', 'morocco', 'tunisia', 'libya', 'north africa']):
        return "Semi-arid"
    elif any(x in loc for x in ['sahara', 'desert', 'sahel', 'egypt']):
        return "Arid"
    elif any(x in loc for x in ['amazon', 'congo', 'rainforest']):
        return "Humid"
    elif any(x in loc for x in ['europe', 'france', 'germany', 'uk', 'spain']):
        return "Humid"
    return "general"

def accuracy_badge_html(level, text):
    css_class = {"high": "accuracy-high", "medium": "accuracy-medium", "low": "accuracy-low"}.get(level, "accuracy-medium")
    return f'<span class="accuracy-badge {css_class}">🎯 {text}</span>'

def generate_climate_data(location, start_year=2023, months=12, region_type="general"):
    """Generate realistic synthetic climate data based on region type."""
    np.random.seed(hash(location) % 2**31)
    
    base_configs = {
        "Arid": {"temp_base": 28, "temp_range": 18, "precip_base": 15, "precip_var": 10},
        "Semi-arid": {"temp_base": 20, "temp_range": 16, "precip_base": 35, "precip_var": 20},
        "Humid": {"temp_base": 14, "temp_range": 10, "precip_base": 80, "precip_var": 30},
        "general": {"temp_base": 18, "temp_range": 12, "precip_base": 55, "precip_var": 25},
    }
    cfg = base_configs.get(region_type, base_configs["general"])
    
    records = []
    for m in range(months):
        month_num = (m % 12) + 1
        seasonal = np.sin((month_num - 3) * np.pi / 6)
        temp = cfg["temp_base"] + cfg["temp_range"] * seasonal * 0.5 + np.random.normal(0, 1)
        precip = max(0, cfg["precip_base"] - cfg["precip_base"] * 0.6 * seasonal + np.random.normal(0, cfg["precip_var"]))
        soil_m1 = max(0.05, 0.2 - seasonal * 0.08 + np.random.normal(0, 0.02))
        soil_m2 = max(0.08, 0.25 - seasonal * 0.06 + np.random.normal(0, 0.015))
        soil_m3 = max(0.1, 0.3 - seasonal * 0.04 + np.random.normal(0, 0.01))
        pet = max(0, temp * 2.2 + 20 + np.random.normal(0, 5))
        records.append({
            "month": month_num,
            "month_name": datetime(start_year, month_num, 1).strftime('%b'),
            "temperature_2m": round(temp, 1),
            "temperature_max": round(temp + np.random.uniform(3, 6), 1),
            "temperature_min": round(temp - np.random.uniform(3, 6), 1),
            "total_precipitation": round(precip, 1),
            "potential_evaporation": round(pet, 1),
            "soil_moisture_0_7cm": round(soil_m1, 3),
            "soil_moisture_7_28cm": round(soil_m2, 3),
            "soil_moisture_28_100cm": round(soil_m3, 3),
        })
    return pd.DataFrame(records)

def generate_vegetation_data(location, index_name, months=24):
    """Generate realistic synthetic vegetation index time series."""
    np.random.seed((hash(location) + hash(index_name)) % 2**31)
    region_type = get_region_type(location)
    
    base_configs = {
        "Arid": {"base": 0.15, "amp": 0.18, "noise": 0.04},
        "Semi-arid": {"base": 0.28, "amp": 0.22, "noise": 0.05},
        "Humid": {"base": 0.55, "amp": 0.20, "noise": 0.06},
        "general": {"base": 0.38, "amp": 0.20, "noise": 0.05},
    }
    cfg = base_configs.get(region_type, base_configs["general"])
    
    # Adjust for different index types
    if index_name in ['NDWI', 'MNDWI', 'AWEI']:
        cfg["base"] -= 0.15
        cfg["amp"] *= 0.8
    elif index_name in ['NBR']:
        cfg["base"] += 0.1
    elif index_name in ['SI', 'NDSI_Salinity']:
        cfg["base"] = 0.4
        cfg["amp"] = 0.15
    
    dates, values = [], []
    start = datetime(2023, 1, 1)
    for i in range(months):
        d = start + timedelta(days=30 * i)
        seasonal = np.sin((d.month - 3) * np.pi / 6)
        val = cfg["base"] + cfg["amp"] * seasonal + np.random.normal(0, cfg["noise"])
        val = max(-1.0, min(1.0, val))
        dates.append(d.strftime('%Y-%m'))
        values.append(round(val, 4))
    return dates, values

def get_soil_data(location, region_type="general"):
    """Generate realistic synthetic soil data."""
    np.random.seed(hash(location) % 2**31)
    
    soil_configs = {
        "Arid": {"texture": 12, "soc": 3.2, "clay": 8, "silt": 12, "sand": 80},
        "Semi-arid": {"texture": 8, "soc": 8.5, "clay": 15, "silt": 20, "sand": 65},
        "Humid": {"texture": 7, "soc": 25.0, "clay": 25, "silt": 40, "sand": 35},
        "general": {"texture": 7, "soc": 15.0, "clay": 20, "silt": 35, "sand": 45},
    }
    cfg = soil_configs.get(region_type, soil_configs["general"])
    
    soc_stock = cfg["soc"] + np.random.normal(0, cfg["soc"] * 0.1)
    soc_stock = max(1.0, soc_stock)
    depth = 20 if region_type in ["Arid", "Semi-arid"] else 30
    soc_pct = soc_stock / (BULK_DENSITY * depth * 100) * 100
    som_pct = soc_pct * SOC_TO_SOM_FACTOR
    
    texture_cls = cfg["texture"]
    clay = cfg["clay"] + np.random.randint(-3, 4)
    silt = cfg["silt"] + np.random.randint(-3, 4)
    sand = 100 - clay - silt
    
    return {
        "texture_class": texture_cls,
        "texture_name": SOIL_TEXTURE_CLASSES.get(texture_cls, "Loam"),
        "soc_stock": round(soc_stock, 2),
        "soil_organic_matter": round(som_pct, 2),
        "bulk_density": BULK_DENSITY,
        "depth_cm": depth,
        "clay_content": max(0, clay),
        "silt_content": max(0, silt),
        "sand_content": max(0, sand),
        "final_som_estimate": round(som_pct, 2),
    }

def get_climate_classification(location, region_type):
    """Return climate classification based on region."""
    np.random.seed(hash(location) % 2**31)
    configs = {
        "Arid": {"zone": "Desert/Steppe", "class": 7, "temp": 28.5, "precip": 120, "aridity": 0.38},
        "Semi-arid": {"zone": "Mediterranean", "class": 6, "temp": 17.8, "precip": 380, "aridity": 1.08},
        "Humid": {"zone": "Oceanic", "class": 8, "temp": 12.5, "precip": 850, "aridity": 2.8},
        "general": {"zone": "Warm Temperate", "class": 9, "temp": 16.0, "precip": 620, "aridity": 1.65},
    }
    cfg = configs.get(region_type, configs["general"])
    return {
        "climate_zone": cfg["zone"],
        "climate_class": cfg["class"],
        "mean_temperature": cfg["temp"] + np.random.normal(0, 0.5),
        "mean_precipitation": cfg["precip"] + np.random.normal(0, 20),
        "aridity_index": cfg["aridity"],
    }

# Step progress display
def progress_bar_html(step, total=4):
    steps_meta = {
        "Climate & Soil": ["Location", "Climate", "Soil", "Results"],
        "Vegetation & Climate": ["Location", "Parameters", "Preview", "Results"],
    }
    labels = steps_meta.get(st.session_state.analysis_type, ["Step1", "Step2", "Step3", "Step4"])
    html = '<div class="progress-container" style="display:flex; gap:0.5rem; margin-bottom:1rem;">'
    for i, label in enumerate(labels, 1):
        if i < step:
            cls = "completed"
            icon = "✓"
        elif i == step:
            cls = "active"
            icon = str(i)
        else:
            cls = ""
            icon = str(i)
        active_color = "#00FF88" if cls == "active" else ("#00CC6A" if cls == "completed" else "#2A2A2A")
        text_col = "#0A0A0A" if cls in ["active", "completed"] else "#999999"
        label_col = "#00FF88" if cls == "active" else "#CCCCCC"
        html += f'''
        <div style="flex:1; display:flex; flex-direction:column; align-items:center;">
            <div style="width:34px;height:34px;border-radius:50%;background:{active_color};display:flex;align-items:center;justify-content:center;color:{text_col};font-weight:700;font-size:0.85rem;margin-bottom:0.4rem;">{icon}</div>
            <span style="font-size:0.7rem;color:{label_col};text-align:center;">{label}</span>
        </div>'''
    html += '</div>'
    return html
